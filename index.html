# Extract year and month
df['year'] = df['date'].dt.year
df['month'] = df['date'].dt.month_name()

# -------------------------------------------
# 1️⃣ Covid-19 Impact: Unemployment during 2020
# -------------------------------------------
covid_df = df[df['year'] == 2020]
covid_monthly_avg = covid_df.groupby(covid_df['date'].dt.to_period('M'))['estimated_unemployment_rate_%'].mean().reset_index()
covid_monthly_avg['date'] = covid_monthly_avg['date'].dt.to_timestamp()

plt.figure(figsize=(12,6))
sns.lineplot(data=covid_monthly_avg, x='date', y='estimated_unemployment_rate_%', marker='o')
plt.title('Monthly Average Unemployment Rate in 2020 (Covid-19 Impact)')
plt.xlabel('Month')
plt.ylabel('Unemployment Rate (%)')
plt.grid(True)
plt.tight_layout()
plt.show()

# -------------------------------------------
# 2️⃣ Yearly Unemployment Trend
# -------------------------------------------
yearly_avg = df.groupby('year')['estimated_unemployment_rate_%'].mean().reset_index()

plt.figure(figsize=(10,5))
sns.barplot(data=yearly_avg, x='year', y='estimated_unemployment_rate_%', palette='magma')
plt.title('Yearly Average Unemployment Rate in India')
plt.ylabel('Unemployment Rate (%)')
plt.xlabel('Year')
plt.tight_layout()
plt.show()

# -------------------------------------------
# 3️⃣ Seasonal (Monthly) Unemployment Trend
# -------------------------------------------
monthly_df = df.copy()
monthly_df['month'] = df['date'].dt.month_name()

monthly_avg = monthly_df.groupby('month')['estimated_unemployment_rate_%'].mean().reset_index()

# Sort months properly
month_order = ['January', 'February', 'March', 'April', 'May', 'June',
               'July', 'August', 'September', 'October', 'November', 'December']
monthly_avg['month'] = pd.Categorical(monthly_avg['month'], categories=month_order, ordered=True)
monthly_avg = monthly_avg.sort_values('month')

plt.figure(figsize=(10,5))
sns.lineplot(data=monthly_avg, x='month', y='estimated_unemployment_rate_%', marker='o')
plt.title('Seasonal Monthly Unemployment Rate in India')
plt.xlabel('Month')
plt.ylabel('Unemployment Rate (%)')
plt.xticks(rotation=45)
plt.grid(True)
plt.tight_layout()
plt.show()
